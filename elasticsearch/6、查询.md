# 6、查询

## 6.1 单个查询

```shell
curl -X PUT "http://localhost:9200/${index}/_search" -d '{
	"query": {
		"match": {
			"{query_field}": "{query_content}"
		}
	}
}'
```

```shell
curl -X PUT "http://localhost:9200/${index}/_search" -d '{
	"query": {
		"match_phrase": {
			"{query_field}": "{query_content}"
		}
	}
}'
```

```shell
curl -X PUT "http://localhost:9200/${index}/_search" -d '{
	"query": {
		"multi_match": {
			"query": "{query_content}",
			"fields": ["{query_field1}", "{query_field2}"]
		}
	}
}'
```

```shell
curl -X PUT "http://localhost:9200/${index}/_search" -d '{
	"query": {
		"query_string": {
			"query": "(XX AND XX) OR XX",
			"fields": ["{query_field1}", "{query_field2}"] # 可选
		}
	}
}'
```

```shell
curl -X PUT "http://localhost:9200/${index}/_search" -d '{
	"query": {
		"term": {
			"{query_field}": "{query_content}"
		}
	}
}'
```

> term 还有点疑问，是完全匹配还是？

```shell
curl -X PUT "http://localhost:9200/${index}/_search" -d '{
	"query": {
		"range": {
			"{query_field}": {
				"gte": 100,
				"lte": 1000
			}
		}
	}
}'
```



## 6.2 Filter

```shell
curl -X POST "http://localhost:9200/${index}/_search" -d '{
	"query": {
		"bool": {
			"filter": {
				"term": {
					"{query_field}": "{query_content}"
				}
			}
		}
	}
}'
```



## 6.3 复合查询

```shell
curl -X PUT "http://localhost:9200/${index}/_search" -d '{
	"query": {
		"bool": {
			"should": {
				{
					"match": {
						"{query_field}": "{query_content}"
					}
				},
				{
					"match": {
						"{query_field}": "{query_content}"
					}
				}
			},
			"must": {
				{
					"match": {
						"{query_field}": "{query_content}"
					}
				},
				{
					"match": {
						"{query_field}": "{query_content}"
					}
				}
			},
			"must_not": {
				{
					"match": {
						"{query_field}": "{query_content}"
					}
				},
				{
					"match": {
						"{query_field}": "{query_content}"
					}
				}
			},
			"filter": {
				"term": {
					"{query_field}": "{query_content}"
				}
			}
			
		}
	}
}'
```

```shell
curl -X GET "http://localhost:9200/${index}/_search" -d {
	"query": {
		"bool": {
			"should": {
				"match": {
					"name": "葡萄酒"
				}
			},
			"filter": {
				"range" : {
					"price": {
						"gte": 100,
						"lte": 200
					}
				}
			}
		}
	}
}
```



## 6.4 排序

```shell
curl -X GET "http://localhost:9200/${index}/_search" -d {
	"query": {
		"bool": {
			"should": {
				"match": {
					"name": "葡萄酒"
				}
			},
			"filter": {
				"range" : {
					"price": {
						"gte": 100,
						"lte": 200
					}
				}
			}
		}
	},
	"sort": [
		{
			"price": {
				"order": "desc"
			}
		}
	]
}
```
